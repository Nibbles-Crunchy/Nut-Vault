// 1. FIXED INITIALIZATION const canvas = document.getElementById('game'); if (canvas) { canvas.width = 1280; canvas.height = 640; } const ctx = canvas.getContext('2d'); const SIZE = 64; let nutStashes = []; // 2. MOUSE TRACKING (Required for KeyR placement) let mousePos = { x: 0, y: 0 }; canvas.addEventListener('mousemove', (e) => { const rect = canvas.getBoundingClientRect(); mousePos.x = e.clientX - rect.left; mousePos.y = e.clientY - rect.top; }); // 3. UPDATED INPUT LOGIC window.addEventListener('keydown', (e) => { if (!user) return; if (e.code === 'KeyR' && data.bal >= 30) { const tx = Math.floor(mousePos.x / SIZE) * SIZE; const ty = Math.floor(mousePos.y / SIZE) * SIZE; // Ensure placement is on the PATH if (PATH.some(p => p.x * SIZE === tx && p.y * SIZE === ty)) { data.bal -= 30; nutStashes.push({ x: tx, y: ty, hp: 100, maxHp: 100 }); updateUI(); } } }); // 4. IMPROVED GAME LOOP function gameLoop() { if (data.hp <= 0) return; ctx.fillStyle = "#2d5a27"; ctx.fillRect(0, 0, 1280, 640); // Draw Path ctx.fillStyle = "rgba(212, 163, 115, 0.8)"; PATH.forEach(p => ctx.fillRect(p.x * SIZE, p.y * SIZE, SIZE, SIZE)); // Draw & Manage Nut Stashes nutStashes = nutStashes.filter(ns => ns.hp > 0); nutStashes.forEach(ns => { ctx.font = "40px Arial"; ctx.fillText("🥜", ns.x + 10, ns.y + 45); // Health bar for the stash ctx.fillStyle = "red"; ctx.fillRect(ns.x + 8, ns.y + 50, 48, 4); ctx.fillStyle = "lime"; ctx.fillRect(ns.x + 8, ns.y + 50, (ns.hp / ns.maxHp) * 48, 4); }); enemies.forEach((en) => { // Collision detection with Stashes let stash = nutStashes.find(n => Math.hypot(n.x + 32 - (en.x + 32), n.y + 32 - (en.y + 32)) < 32 ); if (stash) { stash.hp -= 0.5; // Enemy consumes the stash en.speedMultiplier = 0.2; // Significant slow down } else { en.speedMultiplier = 1.0; // Normal speed } // Apply movement using the multiplier en.x += en.vx * en.speedMultiplier; en.y += en.vy * en.speedMultiplier; // Render Enemy ctx.fillStyle = "red"; ctx.fillRect(en.x, en.y - 16, 64, 12); // Enemy Health Bar ctx.font = en.isBoss ? "60px Arial" : "45px Arial"; ctx.fillText(en.icon, en.x + 8, en.y + 48 + Math.sin(en.bob) * 8); }); requestAnimationFrame(gameLoop); }
